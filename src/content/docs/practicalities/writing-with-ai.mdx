Write C++ code to solve this problem:

Given a list of 1 million random integers between 1 and 100,000, find the difference between the smallest and the largest numbers whose digits sum up to 30.

```cpp
#include <iostream>
#include <vector>
#include <random>
#include <string>
#include <climits>

// Calculate sum of digits for a number
int digitSum(int num) {
    string numStr = to_string(num);
    int sum = 0;
    for (char digit : numStr) {
        sum += (digit - '0');
    }
    return sum;
}

int main() {
    // Initialize random number generator
    random_device rd;
    mt19937 gen(rd());
    uniform_int_distribution<> dis(1, 100000);

    // Generate 1 million random numbers
    vector<int> numbers;
    numbers.reserve(1000000);
    for (int i = 0; i < 1000000; i++) {
        numbers.push_back(dis(gen));
    }

    // Find min and max numbers with digit sum 30
    int min_num = INT_MAX;
    int max_num = INT_MIN;
    
    for (int num : numbers) {
        if (digitSum(num) == 30) {
            min_num = min(min_num, num);
            max_num = max(max_num, num);
        }
    }

    // Calculate and display result
    if (min_num == INT_MAX || max_num == INT_MIN) {
        std::cout << "No numbers found with digit sum of 30" << std::endl;
    } else {
        std::cout << "Difference between largest and smallest numbers with digit sum 30: " 
             << max_num - min_num << std::endl;
        std::cout << "Largest: " << max_num << std::endl;
        std::cout << "Smallest: " << min_num << std::endl;
    }

    return 0;
}
